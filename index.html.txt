        function handleSignup(e){
            e.preventDefault();
            const formContainer = document.getElementById('form-container');
            const successMessage = document.getElementById('successMessage');
            const submitBtn = document.getElementById('submitBtn');
            const formData = new FormData(e.target);
            const data = Object.fromEntries(formData.entries());

            submitBtn.innerHTML = 'Sender...';
            submitBtn.disabled = true;

            const emailBody = `Ny tilmelding til Højeskole - om livet:%0D%0A%0D%0ANavn: ${encodeURIComponent(data.name)}%0D%0AEmail: ${encodeURIComponent(data.email)}%0D%0ATelefon: ${encodeURIComponent(data.phone||'Ikke angivet')}%0D%0AAlder: ${encodeURIComponent(data.age)}%0D%0AInteresse: ${encodeURIComponent(data.interest||'Ikke angivet')}%0D%0ABesked: ${encodeURIComponent(data.message||'Ingen')}%0D%0A%0D%0ATilmeldt: ${new Date().toLocaleString('da-DK')}`;

            setTimeout(()=>{
                window.location.href = `mailto:oejet@hotmail.com?subject=Tilmelding til Højeskole - om livet - ${encodeURIComponent(data.name)}&body=${emailBody}`;
                formContainer.style.display = 'none';
                successMessage.classList.add('active');
                startConfetti();
            }, 1000);
        }

        function startConfetti(){
            const canvas = document.createElement('canvas');
            canvas.style.cssText = 'position:fixed;top:0;left:0;width:100%;height:100%;pointer-events:none;z-index:9999';
            document.body.appendChild(canvas);
            const ctx = canvas.getContext('2d');
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;

            const colors = ['#1B4332', '#2D6A4F', '#52B788', '#D4A373', '#BC6C25', '#E76F51'];
            const pieces = [];
            for(let i=0;i<150;i++){
                pieces.push({
                    x: Math.random()*canvas.width,
                    y: Math.random()*canvas.height - canvas.height,
                    size: Math.random()*10+5,
                    color: colors[Math.floor(Math.random()*colors.length)],
                    speed: Math.random()*4+2,
                    angle: Math.random()*360,
                    rotation: Math.random()*10-5,
                    opacity: Math.random()*0.5+0.5
                });
            }
            let animationId;
            function animate(){
                ctx.clearRect(0,0,canvas.width,canvas.height);
                let active = false;
                pieces.forEach(p=>{
                    p.y += p.speed;
                    p.angle += p.rotation;
                    p.x += Math.sin(p.angle*0.05)*2;
                    if(p.y < canvas.height){
                        active = true;
                        ctx.save();
                        ctx.globalAlpha = p.opacity;
                        ctx.translate(p.x,p.y);
                        ctx.rotate(p.angle*Math.PI/180);
                        ctx.fillStyle = p.color;
                        ctx.fillRect(-p.size/2,-p.size/2,p.size,p.size);
                        ctx.restore();
                    }
                });
                if(active) animationId = requestAnimationFrame(animate);
                else if(canvas.parentNode) canvas.parentNode.removeChild(canvas);
            }
            animate();
            setTimeout(()=>{
                cancelAnimationFrame(animationId);
                if(canvas.parentNode) canvas.parentNode.removeChild(canvas);
            }, 4000);
        }

        const observer = new IntersectionObserver((entries)=>{

                }
            });

            observer.observe(el);
